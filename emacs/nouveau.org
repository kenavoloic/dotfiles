(defun recherche-mot-dico()
  "Recherche le mot sous le curseur dans cnrtl.fr"
  (interactive)
  (let (word)
    (setq word
          (if (use-region-p)
              (buffer-substring-no-properties (region-beginning) (region-end))
            (current-word)))
    (setq word (replace-regexp-in-string " " "_" word))
    (browse-url (concat "http://www.cnrtl.fr/definition/" word))))

(defun plus()
  (interactive)
  (skip-chars-backward "0-9")
  (or (looking-at "[0-9]+")
      (error "No number at point."))
  (replace-match (number-to-string (1+ (string-to-number (match-string 0))))))
(global-set-key (kbd "M-à") 'plus)

(defun moins()
  (interactive)
  (skip-chars-backward "0-9")
  (or (looking-at "[0-9]+")
      (error "No number at point."))
  (replace-match (number-to-string (1- (string-to-number (match-string 0))))))

(use-package flycheck
  :init
  (global-flycheck-mode t)
  :config
  ;; Support intelligent pour uv + ruff
  (defun my/flycheck-find-uv-executable (executable)
    "Find executable using uv or fallback to system PATH."
    (cond
     ;; Essaie d'abord uv si on est dans un projet uv
     ((and (locate-dominating-file default-directory "pyproject.toml")
           (executable-find "uv"))
      (let* ((uv-cmd (format "uv run which %s 2>/dev/null" executable))
             (result (string-trim (shell-command-to-string uv-cmd))))
        (if (and (not (string-empty-p result))
                 (file-exists-p result))
            result
          (executable-find executable))))
     ;; Environnement virtuel classique
     ((getenv "VIRTUAL_ENV")
      (let ((venv-path (expand-file-name 
                       (concat (getenv "VIRTUAL_ENV") "/bin/" executable))))
        (if (file-exists-p venv-path) 
            venv-path 
          (executable-find executable))))
     ;; Fallback système
     (t (executable-find executable))))
  
  ;; Configuration Ruff optimisée pour Django
  (setq flycheck-executable-find #'my/flycheck-find-uv-executable
        flycheck-ruff-args '("--output-format=text" 
                             "--line-length=88"
                             "--extend-ignore=E203,W503"))  ; Compatible avec Black

    ;; Checker personnalisé pour Django
  (defun my/setup-django-flycheck ()
    "Configure flycheck pour les projets Django."
    (when (my/django-project-p)
      ;; Utilise ruff au lieu de flake8 pour Django
      (setq-local flycheck-disabled-checkers 
                  '(python-flake8 python-pycodestyle python-pylint))
      (setq-local flycheck-enabled-checkers 
                  '(python-ruff))))
  
  (add-hook 'python-mode-hook #'my/setup-django-flycheck))

;; RACCOURCIS ESSENTIELS :
;; C-x g   → magit-status
;; C-x G   → magit-dispatch  
;; C-c g c → magit-clone
;; C-c g f → magit-find-file
;; C-c g l → magit-log-buffer-file
;; C-c g b → magit-blame
;; C-c g t → git-timemachine
;;
;; WORKFLOW DJANGO (touche D dans magit-status) :
;; D m → Vérifier migrations manquantes
;; D t → Lancer tests avant commit
;; D d → Template de commit Django
;;
;; GIT-GUTTER TEMPS RÉEL :
;; C-c g n → Hunk suivant
;; C-c g p → Hunk précédent
;; C-c g r → Revert hunk  
;; C-c g s → Stage hunk

;; 1. C-x g → Voir status
;; 2. s → Stage des fichiers

;; 4. D d → Template commit Django
;; 5. c c → Commit avec template
